CREATE TABLE SALESPERSON
(
  sp_id INT NOT NULL,
  name VARCHAR(20) NOT NULL,
  PRIMARY KEY (sp_id)
);

CREATE TABLE CUSTOMER
(
  customer_id INT NOT NULL,
  name VARCHAR(20) NOT NULL,
  phone_no INT NOT NULL,
  PRIMARY KEY (customer_id)
);

CREATE TABLE CARS
(
  serial_no VARCHAR(20) NOT NULL,
  model VARCHAR(20) NOT NULL,
  colour VARCHAR(20) NOT NULL,
  sp_id INT NOT NULL,
  customer_id INT NOT NULL,
  PRIMARY KEY (serial_no),
  FOREIGN KEY (sp_id) REFERENCES SALESPERSON(sp_id),
  FOREIGN KEY (customer_id) REFERENCES CUSTOMER(customer_id)
);

CREATE TABLE INVOICE
(
  sp_id INT NOT NULL,
  customer_id INT NOT NULL,
  PRIMARY KEY (sp_id, customer_id),
  FOREIGN KEY (sp_id) REFERENCES SALESPERSON(sp_id),
  FOREIGN KEY (customer_id) REFERENCES CUSTOMER(customer_id)
);

CREATE TABLE SERVICE_RECORD
(
  service_ticket VARCHAR(20) NOT NULL,
  service_date DATE NOT NULL,
  serial_no VARCHAR(20) NOT NULL,
  PRIMARY KEY (service_ticket),
  FOREIGN KEY (serial_no) REFERENCES CARS(serial_no)
);

CREATE TABLE MECHANICS
(
  mechanic_id INT NOT NULL,
  name VARCHAR(20) NOT NULL,
  PRIMARY KEY (mechanic_id)
);

CREATE TABLE PARTS
(
  parts_name VARCHAR(20) NOT NULL,
  parts_id INT NOT NULL,
  PRIMARY KEY (parts_id)
);

CREATE TABLE WORKS_ON
(
  serial_no VARCHAR(20) NOT NULL,
  mechanic_id INT NOT NULL,
  PRIMARY KEY (serial_no, mechanic_id),
  FOREIGN KEY (serial_no) REFERENCES CARS(serial_no),
  FOREIGN KEY (mechanic_id) REFERENCES MECHANICS(mechanic_id)
);

CREATE TABLE NEEDS
(
  service_ticket VARCHAR(20) NOT NULL,
  parts_id INT NOT NULL,
  PRIMARY KEY (service_ticket, parts_id),
  FOREIGN KEY (service_ticket) REFERENCES SERVICE_RECORD(service_ticket),
  FOREIGN KEY (parts_id) REFERENCES PARTS(parts_id)
);
